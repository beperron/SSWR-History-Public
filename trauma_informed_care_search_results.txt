## KEYWORD SEARCH (SQL)
# The keyword search uses PostgreSQL ILIKE operator for
# case-insensitive pattern matching on paper abstracts.

SQL_QUERY = """
WITH keyword_matches AS (
    -- Match 'trauma-informed care'
    SELECT id, year, title, methodology,
           'trauma-informed care' AS matched_keyword
    FROM papers WHERE abstract ILIKE '%trauma-informed care%'
    UNION ALL
    -- Match 'trauma informed' (space separated)
    SELECT id, year, title, methodology,
           'trauma informed' AS matched_keyword
    FROM papers WHERE abstract ILIKE '%trauma informed%'
    UNION ALL
    -- Match 'trauma-responsive'
    SELECT id, year, title, methodology,
           'trauma-responsive' AS matched_keyword
    FROM papers WHERE abstract ILIKE '%trauma-responsive%'
    UNION ALL
    -- Match 'trauma-sensitive'
    SELECT id, year, title, methodology,
           'trauma-sensitive' AS matched_keyword
    FROM papers WHERE abstract ILIKE '%trauma-sensitive%'
    UNION ALL
    -- Match 'trauma-aware'
    SELECT id, year, title, methodology,
           'trauma-aware' AS matched_keyword
    FROM papers WHERE abstract ILIKE '%trauma-aware%'
)
SELECT DISTINCT ON (id) id, year, title, methodology,
       matched_keyword FROM keyword_matches
ORDER BY id, year DESC LIMIT 20;
"""

import openai
import requests

# --- Configuration ---
OPENROUTER_KEY = 'sk-or-v1-...'  # Your API key
SUPABASE_URL = "https://jomsksqqcpkbuhwytovo.supabase.co"
SUPABASE_KEY = "eyJhbGciOi..."  # Anon key from .claude/skills/
SEMANTIC_MODEL = 'openai/text-embedding-3-small'
SEMANTIC_RPC = 'match_papers_small'
SIMILARITY_THRESHOLD = 0.3
MAX_RESULTS = 20

# --- Step 1: Generate query embedding ---
# Initialize OpenAI client with OpenRouter endpoint
client = openai.OpenAI(
    base_url="https://openrouter.ai/api/v1",
    api_key=OPENROUTER_KEY,
)

# Generate embedding for search query
query_text = "trauma-informed care"
response = client.embeddings.create(
    model=SEMANTIC_MODEL,  # Must match stored embedding model
    input=query_text
)
query_vector = response.data[0].embedding
# Vector dimensions: 1536 for text-embedding-3-small

# --- Step 2: Query Supabase RPC for similar papers ---
# The RPC uses pgvector cosine similarity operator (<=)>
result = requests.post(
    "https://jomsksqqcpkbuhwytovo.supabase.co/rest/v1/rpc/match_papers_small",
    headers={
        "apikey": SUPABASE_KEY,
        "Content-Type": "application/json"
    },
    json={
        'query_embedding': query_vector,
        'match_threshold': SIMILARITY_THRESHOLD,  # Min similarity
        'match_count': MAX_RESULTS,               # Results to return
        'min_year': 2005,                        # Year range filter
        'max_year': 2026
    }
)

results = result.json()

# --- Step 3: Sort by similarity desc, then year desc ---
results.sort(key=lambda x: (-x['similarity'], -x['year']))

# --- Step 4: Display results ---
for i, paper in enumerate(results, 1):
    print(f"{i}. [{paper['year']}] sim={paper['similarity']:.3f}")
    print(f"   {paper['title']}")
    print(f"   {paper['methodology']}")
